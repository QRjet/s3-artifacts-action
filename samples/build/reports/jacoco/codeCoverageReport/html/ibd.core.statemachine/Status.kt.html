package ibd.core.statemachine

enum class Status {
    INITIALIZED,
    PA_NOT_POSSIBLE,
    CHART_NOTES_REQUESTED,
    IN_PROGRESS_VIA_CMM,
    PA_APPROVED_VIA_CMM,
    PA_DENIED_VIA_CMM,
    EXTRA_INFO_REQUESTED,
    ALT_MEDICATION_PROVIDED,
    FILE_AND_CLOSE,
    APPEAL_REQUESTED_VIA_CMM,
    EXTRA_MEDICAL_INFO_REQUESTED,
    APPEAL_IN_PROGRESS_VIA_CMM,
    FOLLOW_UP_CMM,
    APPEAL_APPROVED_VIA_CMM,
    APPEAL_DENIED_VIA_CMM,
    IN_PROGRESS_VIA_FAX,
    PA_APPROVED_VIA_FAX,
    FOLLOW_UP_FAX,
    ADDITIONAL_NOTES_REQUESTED,
    APPEAL_REQUESTED_VIA_FAX,
    APPEAL_IN_PROGRESS_VIA_FAX,
    APPEAL_APPROVED_VIA_FAX;

    fun nextPossibleStatuses(): List<Status> =
        when (this) {
            INITIALIZED -> listOf(IN_PROGRESS_VIA_CMM, IN_PROGRESS_VIA_FAX, PA_NOT_POSSIBLE, CHART_NOTES_REQUESTED)
            IN_PROGRESS_VIA_CMM -> listOf(PA_APPROVED_VIA_CMM, PA_DENIED_VIA_CMM, EXTRA_INFO_REQUESTED)
            CHART_NOTES_REQUESTED -> listOf(IN_PROGRESS_VIA_CMM)
            EXTRA_INFO_REQUESTED -> listOf(IN_PROGRESS_VIA_CMM)
            PA_DENIED_VIA_CMM, APPEAL_DENIED_VIA_CMM -> listOf(ALT_MEDICATION_PROVIDED, FOLLOW_UP_CMM, FILE_AND_CLOSE, APPEAL_REQUESTED_VIA_CMM)
            APPEAL_REQUESTED_VIA_CMM -> listOf(EXTRA_MEDICAL_INFO_REQUESTED, APPEAL_IN_PROGRESS_VIA_CMM)
            EXTRA_MEDICAL_INFO_REQUESTED -> listOf(APPEAL_IN_PROGRESS_VIA_CMM)
            APPEAL_IN_PROGRESS_VIA_CMM -> listOf(APPEAL_APPROVED_VIA_CMM, APPEAL_DENIED_VIA_CMM, EXTRA_INFO_REQUESTED)
            FOLLOW_UP_CMM -> listOf(ALT_MEDICATION_PROVIDED, FILE_AND_CLOSE, APPEAL_REQUESTED_VIA_CMM)
            IN_PROGRESS_VIA_FAX -> listOf(PA_APPROVED_VIA_FAX, FOLLOW_UP_FAX, APPEAL_REQUESTED_VIA_FAX, ALT_MEDICATION_PROVIDED, FILE_AND_CLOSE)
            APPEAL_REQUESTED_VIA_FAX -> listOf(APPEAL_IN_PROGRESS_VIA_FAX, ADDITIONAL_NOTES_REQUESTED)
            ADDITIONAL_NOTES_REQUESTED -> listOf(APPEAL_IN_PROGRESS_VIA_FAX)
            APPEAL_IN_PROGRESS_VIA_FAX -> listOf(APPEAL_APPROVED_VIA_FAX, FOLLOW_UP_FAX, ALT_MEDICATION_PROVIDED, FILE_AND_CLOSE)
            FOLLOW_UP_FAX -> listOf(ALT_MEDICATION_PROVIDED, FILE_AND_CLOSE, PA_APPROVED_VIA_FAX, APPEAL_REQUESTED_VIA_FAX)
            else -> emptyList()
        }
}
