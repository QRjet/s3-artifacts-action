package task

import java.sql.PreparedStatement
import java.sql.Timestamp
import kotlin.Int
import kotlin.String
import norm.Command
import norm.ParamSetter

data class UpdateTaskClaimParams(
  val userName: String?,
  val claimDate: Timestamp?,
  val status: String?,
  val id: Int?
)

class UpdateTaskClaimParamSetter : ParamSetter<UpdateTaskClaimParams> {
  override fun map(ps: PreparedStatement, params: UpdateTaskClaimParams) {
    ps.setObject(1, params.userName)
    ps.setObject(2, params.claimDate)
    ps.setObject(3, params.status)
    ps.setObject(4, params.id)
  }
}

class UpdateTaskClaimCommand : Command<UpdateTaskClaimParams> {
  override val sql: String = """
      |update pa_tasks p  set user_name=?, task_claimed_at=?, status=COALESCE(status, ?) where p.id=?;
      |""".trimMargin()

  override val paramSetter: ParamSetter<UpdateTaskClaimParams> = UpdateTaskClaimParamSetter()
}
