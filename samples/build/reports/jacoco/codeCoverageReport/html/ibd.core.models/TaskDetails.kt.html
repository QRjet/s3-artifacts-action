package ibd.core.models

import ibd.core.statemachine.Status
import ibd.core.users.UserBasicDetails
import java.time.OffsetDateTime
import java.util.*

data class TaskDetails(
    val patient: Patient,
    val insurance: Insurance,
    val doctor: Doctor,
    val prescription: Prescription,
    val additionalInformation: AdditionalInformation,
    val status: Status?,
    val timestampForLatestStatusUpdate: OffsetDateTime?,
    val nextPossibleStatuses: List<Status>?,
    val claimedBy: UserBasicDetails?,
    val type: TaskType
)

data class Patient(
    val firstName: String,
    val lastName: String,
    val dob: Date?,
    val gender: String,
    val email: String?,
    val mobile: String?,
    val telephone: String?,
    val address: Address?
)

data class Address(val line1: String, val line2: String, val city: String, val state: String, val zipcode: String)

data class Insurance(
    val id: String?,
    val bin: String?,
    val pcn: String?,
    val groupNo: String?,
    val companyName: String?,
    val type: String?
)

data class Doctor(
    val npi: String,
    val firstName: String,
    val lastName: String,
    val office: String?,
    val email: String?,
    val faxNo: String?,
    val address: Address?
)

data class Prescription(
    val rxNo: String,
    val createdAt: Date?,
    val drugName: String,
    val quantity: Double?,
    val refillNumber: Int,
    val refills: Int,
    val direction: String?,
    val daySupply: Int?,
    val amount: Double,
    val awp: Double?,
    val drugForm: String?,
    val diagnosisCode: String?,
    val ndc: String
)
