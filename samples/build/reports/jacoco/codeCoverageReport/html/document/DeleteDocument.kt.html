package document

import java.sql.PreparedStatement
import kotlin.Int
import kotlin.String
import norm.Command
import norm.ParamSetter

data class DeleteDocumentParams(
  val id: Int?
)

class DeleteDocumentParamSetter : ParamSetter<DeleteDocumentParams> {
  override fun map(ps: PreparedStatement, params: DeleteDocumentParams) {
    ps.setObject(1, params.id)
  }
}

class DeleteDocumentCommand : Command<DeleteDocumentParams> {
  override val sql: String = "delete from documents where id = ?"

  override val paramSetter: ParamSetter<DeleteDocumentParams> = DeleteDocumentParamSetter()
}
