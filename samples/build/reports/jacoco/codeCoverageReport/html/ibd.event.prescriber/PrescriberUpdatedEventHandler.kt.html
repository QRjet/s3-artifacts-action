package ibd.event.prescriber

import ibd.connector.db.prescription.Address
import ibd.connector.db.prescription.Doctor
import ibd.connector.db.prescription.PrescriptionRepository
import ibd.event.EventHandler
import javax.inject.Inject

class PrescriberUpdatedEventHandler(@Inject private val prescriptionRepository: PrescriptionRepository) : EventHandler<PrescriberUpdatedEvent> {
    override fun handle(event: PrescriberUpdatedEvent, pharmacyId: String) =
        prescriptionRepository.updateDoctor(
            Doctor(event.mmsId, event.npi, event.firstName, event.lastName, event.mobile, event.email, event.fax, Address(event.address.line1, event.address.line2, event.address.city, event.address.state, event.address.zipcode)),
            pharmacyId)
}
