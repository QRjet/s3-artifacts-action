package ibd.api.tasks.models

import com.fasterxml.jackson.annotation.JsonFormat
import com.fasterxml.jackson.annotation.JsonInclude
import ibd.core.models.TaskType
import ibd.core.statemachine.Status
import java.math.BigDecimal
import java.util.*

data class TaskResponse(
		val id: Int,
		val rxNumber: String,
		val patientFirstName: String,
		val patientLastName: String,
		val patientDob: String,
		val drugName: String,
		val fillNumber: Int,
		val refills: Int,
		val amount: BigDecimal,
		val doctorFirstName: String,
		val doctorLastName: String,
		val timestamp: String,
		val insuranceCode: String?,
		val type: String
)

data class TaskDetailsResponse(
		val patient: Patient,
		val insurance: Insurance,
		val doctor: Doctor,
		val prescription: Prescription,
		val additionalInformation: AdditionalInformation,
		val status: Status?,
		@JsonInclude(JsonInclude.Include.ALWAYS) val documents: List<Document>,
		val daysInProgress: Long,
		val nextPossibleStatuses: List<Status>?,
		val claimedBy: User?,
		val hasClaimAccess: Boolean
)

data class User(val userName: String, val fullName: String)

data class AdditionalInformation(
		val keycode: String?,
		val caseId: String?,
		val pbmCompanyName: String?,
		val paNotPossibleReason: String?,
		val paNotPossibleNote: String?,
		val paNumber: String?,
		@JsonFormat(shape = JsonFormat.Shape.STRING, pattern = "MM/dd/yyyy") val paApprovedDate: Date?,
		val extraInfoRequestedType: String?,
		val extraInfoRequestedNote: String?,
		val extraMedicalInfoRequestedType: String?,
		val extraMedicalInfoRequestedNote: String?,
		val paDeniedReferenceNumber: String?,
		@JsonFormat(shape = JsonFormat.Shape.STRING, pattern = "MM/dd/yyyy") val paDeniedDate: Date?,
		val paDeniedReason: String?,
		val paDeniedNote: String?,
		val altMedicationDrugName: String?,
		val altMedicationSig: String?,
		val altMedicationQuantity: Double?,
		val altMedicationRefills: Int?
)

data class Patient(
		val firstName: String,
		val lastName: String,
		val dob: String?,
		val age: Int?,
		val gender: String,
		val email: String?,
		val mobile: String?,
		val telephone: String?,
		val address: Address?
)

data class Address(val line1: String, val line2: String, val city: String, val state: String, val zipcode: String)

data class Insurance(
		val id: String?,
		val bin: String?,
		val pcn: String?,
		val groupNo: String?,
		val companyName: String?,
		val type: String?
)

data class Doctor(
		val npi: String,
		val firstName: String,
		val lastName: String,
		val office: String?,
		val email: String?,
		val faxNo: String?,
		val address: Address?
)


data class Prescription(
		val rxNo: String,
		val createdAt: String?,
		val drugName: String,
		val quantity: Double,
		val refillNumber: Int,
		val refills: Int,
		val direction: String?,
		val daySupply: Int?,
		val amount: Double,
		val awp: Double?,
		val drugForm: String?,
		val diagnosisCode: String?,
		val ndc: String
)

data class TaskStatusUpdateRequest(
		val status: Status?,
		val additionalInformation: AdditionalInformation
)

data class Document(
		val id: Int, val name: String, val uploadedBy: String, val status: Status, val uploadedAt: String, val canDelete: Boolean
)

data class TaskOverviewResponse(
	val associateName: String?,
	val associateRole: String?,
	val drugName: String,
	val rxNo: String,
	val status: String?,
	val taskType: String?,
	val daysInProgress: Int
)

data class TaskHistoryNodeResponse(
	val additionalInformation: AdditionalInformation?,
	val status: Status?,
	val userName: String?,
	val type: TaskType?,
	val eventTimestamp: String,
	val documents: List<Document>
)

data class Response<T>(
		val status: ResponseType,
		val message: String?,
		val body: T?,
		val headers: Map<CharSequence, CharSequence> = emptyMap()
)

enum class ResponseType {
	SUCCESS, BAD_REQUEST, NOT_FOUND, FORBIDDEN;
}
