package prescription

import java.sql.Date
import java.sql.PreparedStatement
import kotlin.Int
import kotlin.String
import norm.Command
import norm.ParamSetter
import org.postgresql.util.PGobject

data class UpdatePatientParams(
  val patientFirstName: String?,
  val patientLastName: String?,
  val patientDob: Date?,
  val patientGender: String?,
  val patientEmail: String?,
  val patientMobile: String?,
  val patientTelephone: String?,
  val patientAddress: PGobject?,
  val patientMmsId: Int?,
  val pharmacyId: String?
)

class UpdatePatientParamSetter : ParamSetter<UpdatePatientParams> {
  override fun map(ps: PreparedStatement, params: UpdatePatientParams) {
    ps.setObject(1, params.patientFirstName)
    ps.setObject(2, params.patientLastName)
    ps.setObject(3, params.patientDob)
    ps.setObject(4, params.patientGender)
    ps.setObject(5, params.patientEmail)
    ps.setObject(6, params.patientMobile)
    ps.setObject(7, params.patientTelephone)
    ps.setObject(8, params.patientAddress)
    ps.setObject(9, params.patientMmsId)
    ps.setObject(10, params.pharmacyId)
  }
}

class UpdatePatientCommand : Command<UpdatePatientParams> {
  override val sql: String = """
      |update prescriptions set patient_first_name=?, patient_last_name=?, patient_dob=?, patient_gender=?, patient_email=?, patient_mobile=?, patient_telephone=?, patient_address=?
      |where patient_mms_id = ? and pharmacy_id = ?
      """.trimMargin()

  override val paramSetter: ParamSetter<UpdatePatientParams> = UpdatePatientParamSetter()
}
