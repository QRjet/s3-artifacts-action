package prescription

import java.math.BigDecimal
import java.sql.PreparedStatement
import kotlin.Int
import kotlin.String
import norm.Command
import norm.ParamSetter

data class UpdatePrescriptionParams(
  val drugName: String?,
  val ndc: String?,
  val directions: String?,
  val daysSupply: Int?,
  val drugForm: String?,
  val diagnosisCode: String?,
  val refills: Int?,
  val quantity: BigDecimal?,
  val amount: BigDecimal?,
  val awp: BigDecimal?,
  val copay: BigDecimal?,
  val discount: BigDecimal?,
  val insuranceId: String?,
  val bin: String?,
  val pcn: String?,
  val groupNumber: String?,
  val insuranceName: String?,
  val insuranceType: String?,
  val rxNumber: String?,
  val fillNumber: Int?
)

class UpdatePrescriptionParamSetter : ParamSetter<UpdatePrescriptionParams> {
  override fun map(ps: PreparedStatement, params: UpdatePrescriptionParams) {
    ps.setObject(1, params.drugName)
    ps.setObject(2, params.ndc)
    ps.setObject(3, params.directions)
    ps.setObject(4, params.daysSupply)
    ps.setObject(5, params.drugForm)
    ps.setObject(6, params.diagnosisCode)
    ps.setObject(7, params.refills)
    ps.setObject(8, params.quantity)
    ps.setObject(9, params.amount)
    ps.setObject(10, params.awp)
    ps.setObject(11, params.copay)
    ps.setObject(12, params.discount)
    ps.setObject(13, params.insuranceId)
    ps.setObject(14, params.bin)
    ps.setObject(15, params.pcn)
    ps.setObject(16, params.groupNumber)
    ps.setObject(17, params.insuranceName)
    ps.setObject(18, params.insuranceType)
    ps.setObject(19, params.rxNumber)
    ps.setObject(20, params.fillNumber)
  }
}

class UpdatePrescriptionCommand : Command<UpdatePrescriptionParams> {
  override val sql: String = """
      |update prescriptions set drug_name=?, ndc=?, directions=?, days_supply=?, drug_form=?, diagnosis_code=?, refills=?, quantity=?, amount=?, awp=?, copay=?, discount=?, insurance_id=?, bin=?, pcn=?, group_number=?, insurance_name=?, insurance_type=?
      |where rx_number = ? and fill_number = ?
      |""".trimMargin()

  override val paramSetter: ParamSetter<UpdatePrescriptionParams> = UpdatePrescriptionParamSetter()
}
