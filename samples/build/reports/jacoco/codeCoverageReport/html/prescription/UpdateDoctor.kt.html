package prescription

import java.sql.PreparedStatement
import kotlin.Int
import kotlin.String
import norm.Command
import norm.ParamSetter
import org.postgresql.util.PGobject

data class UpdateDoctorParams(
  val doctorNpi: String?,
  val doctorFirstName: String?,
  val doctorLastName: String?,
  val doctorMobile: String?,
  val doctorEmail: String?,
  val doctorFax: String?,
  val doctorAddress: PGobject?,
  val doctorMmsId: Int?,
  val pharmacyId: String?
)

class UpdateDoctorParamSetter : ParamSetter<UpdateDoctorParams> {
  override fun map(ps: PreparedStatement, params: UpdateDoctorParams) {
    ps.setObject(1, params.doctorNpi)
    ps.setObject(2, params.doctorFirstName)
    ps.setObject(3, params.doctorLastName)
    ps.setObject(4, params.doctorMobile)
    ps.setObject(5, params.doctorEmail)
    ps.setObject(6, params.doctorFax)
    ps.setObject(7, params.doctorAddress)
    ps.setObject(8, params.doctorMmsId)
    ps.setObject(9, params.pharmacyId)
  }
}

class UpdateDoctorCommand : Command<UpdateDoctorParams> {
  override val sql: String = """
      |update prescriptions set doctor_npi=?, doctor_first_name=?, doctor_last_name=?, doctor_mobile=?, doctor_email=?, doctor_fax=?, doctor_address=?
      |where doctor_mms_id = ? and pharmacy_id = ?
      """.trimMargin()

  override val paramSetter: ParamSetter<UpdateDoctorParams> = UpdateDoctorParamSetter()
}
