package ibd.core.datetime

import java.time.*
import java.util.*
import java.util.concurrent.TimeUnit
import javax.inject.Singleton
import kotlin.math.abs

@Singleton
class DateProvider {
    private val zoneId = ZoneOffset.UTC
    fun getCurrentDateTime(): OffsetDateTime = OffsetDateTime.ofInstant(Instant.now(), zoneId)
    fun getDifferenceInYears(offsetDateTime1: OffsetDateTime, offsetDateTime2: OffsetDateTime) =
        Period.between(offsetDateTime2.toLocalDate(), offsetDateTime1.toLocalDate()).years

    fun getDifferenceInDays(offsetDateTime1: OffsetDateTime, offsetDateTime2: OffsetDateTime) =
        TimeUnit.DAYS.convert(abs(offsetDateTime1.toEpochSecond() - offsetDateTime2.toEpochSecond()), TimeUnit.SECONDS)

    fun dateToOffsetDateTime(date: Date): OffsetDateTime = OffsetDateTime.ofInstant(Instant.ofEpochMilli(date.time), zoneId)
}
