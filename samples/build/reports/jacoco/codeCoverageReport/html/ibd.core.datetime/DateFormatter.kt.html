package ibd.core.datetime

import java.text.SimpleDateFormat
import java.time.OffsetDateTime
import java.time.format.DateTimeFormatter
import java.util.*
import javax.inject.Singleton

@Singleton
class DateFormatter {
    fun formatDate(date: Date): String = SimpleDateFormat("MM-dd-yyyy").format(date)
    fun formatDate(date: OffsetDateTime) = date.format(DateTimeFormatter.ISO_ZONED_DATE_TIME)
}
