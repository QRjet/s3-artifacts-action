package ibd.scheduledEvent

import com.amazonaws.services.lambda.runtime.events.ScheduledEvent
import ibd.core.alerts.AlertRepository
import io.micronaut.function.FunctionBean
import io.micronaut.function.aws.MicronautRequestHandler
import java.util.function.Consumer
import javax.inject.Inject

@FunctionBean("scheduled-event-listener")
class ScheduledEventListener : MicronautRequestHandler<ScheduledEvent, Unit>(), Consumer<ScheduledEvent> {
    @Inject
    lateinit var alertRepository: AlertRepository

    override fun accept(scheduledEvent: ScheduledEvent) {
        execute(scheduledEvent)
    }

    override fun execute(input: ScheduledEvent?) {
        alertRepository.addAlertsForOverdueTasks()
    }
}
