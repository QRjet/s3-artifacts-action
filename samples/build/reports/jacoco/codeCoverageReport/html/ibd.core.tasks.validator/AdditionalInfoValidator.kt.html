package ibd.core.tasks.validator

import ibd.core.errors.Error
import ibd.core.models.AdditionalInformation
import ibd.core.statemachine.Status
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class AdditionalInfoValidator(@Inject private val validationStrategies: List<AdditionalInfoValidationStrategy>) {

    fun validate(status: Status?, additionalInformation: AdditionalInformation): Error? {
        if (status == null) return Error("Status cannot be null")
        return validationStrategies.firstOrNull { it.status() == status }?.validate(additionalInformation)
    }
}
