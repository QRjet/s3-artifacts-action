package ibd.core.tasks.validator

import ibd.core.errors.Error
import ibd.core.models.AdditionalInformation
import ibd.core.statemachine.Status
import javax.inject.Singleton

@Singleton
class PANotPossibleValidationStrategy : AdditionalInfoValidationStrategy {
    override fun status(): Status = Status.PA_NOT_POSSIBLE
    override fun validate(additionalInformation: AdditionalInformation): Error? =
        if (additionalInformation.paNotPossibleReason == null) Error("Reason for PA not being possible is required") else null
}
