package ibd.core.tasks.validator

import ibd.core.errors.Error
import ibd.core.models.AdditionalInformation
import ibd.core.statemachine.Status
import javax.inject.Singleton

@Singleton
class PADeniedValidationStrategy : AdditionalInfoValidationStrategy {
    override fun status(): Status = Status.PA_DENIED_VIA_CMM
    override fun validate(additionalInformation: AdditionalInformation): Error? {
        if (additionalInformation.paDeniedReferenceNumber == null) return null
        if (additionalInformation.paDeniedReferenceNumber.length > 10 || additionalInformation.paDeniedReferenceNumber.length < 6) return Error("Reference number for PA denied should be between 6 and 10")
        if (!additionalInformation.paDeniedReferenceNumber.matches("[0-9]+".toRegex())) return Error("Reference number for PA denied should have only numeric characters")
        return null
    }
}
