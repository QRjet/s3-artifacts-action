package ibd.core.tasks.validator

import ibd.core.errors.Error
import ibd.core.models.AdditionalInformation
import ibd.core.statemachine.Status
import javax.inject.Singleton

@Singleton
class InProgressViaFaxValidationStrategy : AdditionalInfoValidationStrategy {

    override fun status(): Status = Status.IN_PROGRESS_VIA_FAX
    override fun validate(additionalInformation: AdditionalInformation) = if (!additionalInformation.pbmCompanyName.isNullOrEmpty() && !additionalInformation.pbmCompanyName.matches("[A-Za-z][A-Za-z ]+".toRegex())) Error("PBM company name can only have alphabets") else null
}
