package ibd.core.alerts

enum class AlertType(private val value: String) {
    PA_NOT_POSSIBLE("PA not possible alert"), OVERDUE("Overdue alert");

    companion object {
        fun from(alertType: String) = values().firstOrNull { it.value == alertType }
    }
}
