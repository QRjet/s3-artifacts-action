package ibd.core.alerts

import ibd.core.exceptions.AccessDeniedException
import ibd.core.models.Alert
import ibd.core.roles.Roles
import ibd.core.users.UserRepository
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class AlertService(
    @Inject private val alertRepository: AlertRepository,
    @Inject private val userRepository: UserRepository
) {
    fun getAlerts(resolveStatus: String?, page: Int, limit: Int, userName: String): List<Alert> {
        if (userRepository.getUser(userName).role == Roles.MANAGER)
            return alertRepository.getAlerts(resolveStatus, page * limit, limit)
        else throw AccessDeniedException("Only managers are allowed to access this")
    }
}
