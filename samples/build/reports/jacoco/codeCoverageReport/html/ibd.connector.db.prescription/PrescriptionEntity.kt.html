package ibd.connector.db.prescription

import java.util.*

data class PrescriptionEntity(
    val id: Long?,
    val patient: Patient,
    val doctor: Doctor,
    val prescriptionDetails: PrescriptionDetails,
    val insurance: Insurance
)

data class Patient(
    val mmsUserId: Int,
    val firstName: String,
    val lastName: String,
    val dob: Date,
    val gender: String,
    val email: String?,
    val mobile: String?,
    val telephone: String?,
    val address: Address
)

data class Address(val line1: String?, val line2: String?, val city: String?, val state: String?, val zipcode: String?)

data class PrescriptionDetails(
    val rxNo: String,
    val createdTimestamp: Date,
    val drugName: String,
    val followUpTag: String?,
    val copay: Double,
    val amount: Double,
    val awp: Double?,
    val ndc: String,
    val quantity: Double?,
    val refillNumber: Int,
    val refills: Int,
    val direction: String?,
    val daySupply: Int?,
    val discount: Double?,
    val drugForm: String?,
    val diagnosisCode: String?
)

data class Insurance(
    val id: String?,
    val bin: String?,
    val pcn: String?,
    val groupNo: String?,
    val companyName: String?,
    val type: String?
)

data class Doctor(
    val mmsId: Int,
    val npi: String,
    val firstName: String,
    val lastName: String,
    val mobile: String?,
    val email: String?,
    val faxNo: String?,
    val address: Address
)
