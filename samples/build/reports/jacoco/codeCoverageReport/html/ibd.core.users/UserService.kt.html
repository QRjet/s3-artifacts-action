package ibd.core.users

import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class UserService(@Inject private val userRepository: UserRepository) {
    fun create(user: UserCreatePayload): UserBasicDetails = userRepository.save(user)
}

interface UserRepository {
    fun save(user: UserCreatePayload): UserBasicDetails
    fun getUser(userName: String): User
}
